💻 API 调用示例代码

Python 示例

from openai import OpenAI

client = OpenAI(
    base_url="https://zzzzapi.com/v1",
    api_key="sk-xxxx"
)
stream = client.chat.completions.create(
    model="gemini-pro",
    messages=[{"role": "user", "content": "介绍一下 Gemini 模型的能力"}],
    stream=True,
)
for chunk in stream:
    if chunk.choices[0].delta.content is not None:
        print(chunk.choices[0].delta.content, end="")

JavaScript 示例

import OpenAI from 'openai';

const openai = new OpenAI({
  baseURL: 'https://zzzzapi.com/v1',
  apiKey: 'sk-xxxx',
});

async function callGemini() {
  const stream = await openai.chat.completions.create({
    model: 'gemini-pro',
    messages: [{ role: 'user', content: '介绍一下 Gemini 模型的能力' }],
    stream: true,
  });
  
  for await (const chunk of stream) {
    process.stdout.write(chunk.choices[0]?.delta?.content || '');
  }
}

callGemini();

Java 示例

import com.theokanning.openai.OpenAiApi;
import com.theokanning.openai.completion.chat.*;
import com.theokanning.openai.service.OpenAiService;

import java.time.Duration;
import java.util.ArrayList;
import java.util.List;

public class GeminiExample {
    public static void main(String[] args) {
        String token = "sk-xxxx";
        OpenAiService service = new OpenAiService(token, Duration.ofSeconds(30));
        
        List messages = new ArrayList<>();
        messages.add(new ChatMessage(ChatMessageRole.USER.value(), "介绍一下 Gemini 模型的能力"));
        
        ChatCompletionRequest request = ChatCompletionRequest.builder()
            .model("gemini-pro")
            .messages(messages)
            .build();
            
        service.createChatCompletion(request).getChoices().forEach(choice -> {
            System.out.println(choice.getMessage().getContent());
        });
    }
}
